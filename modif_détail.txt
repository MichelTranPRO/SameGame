public ObjetGrille(int y, int x, Color clr, MaMatrice m, PanelScore s){
        super();
        this.x = x;
        this.y = y;
        this.setClr(clr);
        this.matriceRef = m;
        this.scorePanel = s; 

        this.addMouseListener(new MouseAdapter(){
            @Override
            public void mouseClicked(MouseEvent e){
                int nbVoisin;
                ObjetGrille[][] matrice = matriceRef.getMatr(); 
                ObjetGrille[] voisins = AlgoVoisin.getVoisin(matrice, y, x); // on attrape tous les voisins
                if (voisins[1] != null){
                    for (nbVoisin = 0; voisins[nbVoisin] != null; nbVoisin++) {
                    }
                    for (ObjetGrille obj : voisins){
                        if (obj != null){
                            obj.setVisible(false); // on le cache
                            matrice[obj.getMyY()][obj.getMyX()] = null; // on l'enlève de la matrice 
                        }
                    }
                    scorePanel.setScore(nbVoisin); // ajout du nouveau score
                }
            }
        });
    }

public MaMatrice(int hauteur, int largeur, PanelJeu jeu, char[][] couleur, PanelScore scorePanel){ // modif icis
		super();
		this.grilleTemp = new char[10][15];
		for (int i = 0; i < 10; i++){
			for (int j = 0; j < 15; j++){
				this.grilleTemp[i][j] = couleur[i][j];
			}
		}
		this.matr = new ObjetGrille[hauteur][largeur];
		for(int ligne = 0; ligne < hauteur; ligne += 1){
			for(int col = 0; col < largeur; col += 1){
				if (couleur[ligne][col] == 'R'){
					couleurObjet = Color.RED;
				}
				else if (couleur[ligne][col] == 'V'){
					couleurObjet = Color.GREEN;
				}
				else if (couleur[ligne][col] == 'B'){
					couleurObjet = Color.CYAN;
				}
				this.matr[ligne][col] = new ObjetGrille(ligne, col, couleurObjet, this, scorePanel); // modif ici
				jeu.add(matr[ligne][col]);
			}
		}
	}
